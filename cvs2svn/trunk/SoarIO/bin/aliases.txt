# Default aliases
#
# This essentially adds commands to the command set
# if the first word of a command line == the last word of an alias,
# that word is replaced by the rest of the alias line
#
# Example:
# > step -d
# becomes:
# > run 1 -d
# with alias:
# run 1 step
#
# Format: substitution words [...] command_to_substitute
#
alias -d unalias
cd chdir
firing-counts fc
gds-print gds_print
help man
help ?
init-soar init
ls dir
quit exit
run 1 step
run -d 1 d
run -e 1 e
stop-soar stop
stop-soar interrupt
watch w
